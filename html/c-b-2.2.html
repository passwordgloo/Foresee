    • 17.1求素数
    #include <stdio.h>
#include 
int main(void)
{
	int a;
	printf("请输入一个整数:");
	scanf("%d",&a);
	if(a<=1)
		printf("这个数不是素数\n");
	else if(a==2)
		printf("这个数是素数\n");
	else
	{
		double b =sqrt(a);        //sqrt函数定义在math.h里面，求一个数平方根
		for(int i=2;i<=b;++i)
		{
			if(a%i ==0)
			{
				printf("这个数不是素数\n");
				return 0;
			}
		}
			printf("这个数是素数\n");
	}
	return 0;
}
