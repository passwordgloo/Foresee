JAVA笔记

变量优化

  • 优化前
class Var01{
	public static void main(String[] args){        //main方法相当于入口 
		int a;					//第一步：声明 
		double b;
		String c;
		double d;
		a=34;                                 //第二步：赋值 
		b=30/2;
		c="欢迎";
		d=a+b;
		System.out.println(a);                 //第三步：输出 
		System.out.println(b);
		System.out.println(c);
		System.out.println(d);
	}
}
  • 第一次优化
class Var02{
	public static void main(String[] args){  
		int a = 1+2;                           //数据类型 变量名 赋值 
		double b = 0.25;
		String c = "最强大脑";
		double d = a+b;
		System.out.println("a = "+a);          //+表示连接左右两边 
		System.out.println("b = "+b);
		System.out.println("c = "+c);
		System.out.println("d = "+d);
	}
}
