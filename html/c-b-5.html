    • 32.do……while语句
    #include <stdio.h>
int main(void)
{
	int i=1,s = 1,sum =0;
	do
	{
		s=s*i;
		sum=sum+s;
		++i;                //思考i++与++i
	}
	while(i<=5);            //思考while条件
	printf("%d\n",sum);
	return 0;
}
    • 33.break语句
    #include <stdio.h>
int main(void)
{
	int a,sum =0;
	for(int i =1;i <=10;++i)
	{
		scanf("%d",&a);
		sum+=a;
		if(sum>=1000)
			break;     //break跳出循环语句
	}
	printf("总的捐款数目为:%d\n",sum);
	return 0;
}

    • 37.循环习题3
    #include <stdio.h>
int main(void)
{
	char a,b,c;
	for(char i ='x';i<='z';++i)
	{
		a=i;
		for(char j='x';j<='z';++j)
		{
			if(i==j)
				continue;
			b=j;
			for(char k='x';k<='z';++k)
			{
				if(k==i || k==j)
					continue;
				c=k;
				if(a!='X' && c!='X' && c!='Z')
					printf("Aand%c,Band%c,Cand%ccompete\n",a,b,c);
			}
		}
	}
	return 0;
}
